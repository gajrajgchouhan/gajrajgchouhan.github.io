from matplotlib import pyplot as plt
from math import exp
from random import randint
from numpy import median

def likely_func(func):
    def wrapper(numbers, *args):
        s = 1
        for i in numbers:
            s *= func(i, *args)
        return s
    return wrapper

def double_exp(x, theta):
    return (1/2) * (exp(-abs(x - theta)))

X = 5, 10, 15, 20, 25, 30

for x in X:
    y = []
    numbers = sorted([randint(1, x) for i in range(x)])
    # for distinct values of i (or xi) in range (1, n), median will be equal to middle element.
    likelihood_function = likely_func(double_exp)
    for n in numbers:
        y.append(likelihood_function(numbers, n))
    print(f"Median of numbers = {median(numbers)}")
    for n, _y in zip(numbers, y):
        plt.scatter(n, _y, color='blue')
    plt.xticks(numbers, numbers)
    plt.title(f"X = {x}")
    plt.show()
